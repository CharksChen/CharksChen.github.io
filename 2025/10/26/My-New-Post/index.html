快速幂的代码

#include<stdio.h>
int power(int a,int b,int p)
{
int r=1;
while(b)
{
if(b&1) r=(ra%p);
a=aa%p;
b>>=1;
}
return r;
}
int main()
{
int a,b,s,p;
scanf(“%d %d %d”,&a,&b,&p);
s=power(a,b,p);
printf(“%d^%d mod %d=%d”,a,b,p,s);
return 0;
}
